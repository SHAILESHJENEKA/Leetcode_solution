class Solution {
  public:
    bool isPalindrome(string s) {
      string temp = "";

      for (int i = 0; i < s.length(); i++) {
        if (isalnum(s[i])) {
          temp.push_back(s[i]);
        }
      }
      transform(temp.begin(), temp.end(), temp.begin(), ::tolower);
      string x = temp;

      reverse(temp.begin(), temp.end());
      std::cout << temp;
      if (temp == x) {
        return true;
      } else {
        return false;
      }

    }
};
