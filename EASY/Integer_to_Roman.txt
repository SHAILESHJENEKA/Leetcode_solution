class Solution {
public:
    int romanToInt(string s) {
        map<char, int> m;
        
        int len = s.length();
        int sum=0, num=0;
        
        for(int i=0; i<len; ){
            if(i==(len-1)   || m[s[i]]>=m[s[i+1]]){
                num = m[s[i]];
                i++;
            }else{
                num = m[s[i+1]] - m[s[i]];
                i = i+2;
            }
            sum = sum + num;
        }
        return sum;        
    }
};
